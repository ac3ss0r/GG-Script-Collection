#include<iostream>

// decimal to hex converter by nullBit



std::string decToHex(int n) {
    int temp;
    char arr[10000];
    int z = 0;
    std::string hex = "";
    while(n != 0) {
        temp = n % 16;
        if(temp < 10) {
            arr[z] = 48 + temp;
        }
        else if(temp >= 10) {
            arr[z] = 55 + temp;
        }
        else {
            std::cout << "Something went wrong";
            break;
        }
        n = n/16;
        hex = arr[z] + hex;
        z++;
        
    }
    
    return hex;
      
}



int main()
{
    int user;
    while(true) {
        std::cout << "Input a decimal number: ";
        std::cin >> user;
        std::cout << "Hex is " << decToHex(user) << "\n" ;
    }
    return 0;
}
