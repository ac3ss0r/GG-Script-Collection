function choosemode()
	mode = 1
end

function unlock(id, alert)
	edit = 1
	if id == nil then
		success = 0
		edit = 0
	end
	if gg.getResultsCount() > 14 then
		success = 0
		edit = 0
	end
	gg.setVisible(false)
	gg.setRanges(gg.REGION_ANONYMOUS)
	gg.clearResults()
	gg.searchNumber(id, gg.TYPE_DWORD)
	success = 1
	if edit == 1 then
		if gg.getResultsCount() == 4 then
			gg.getResults(gg.getResultsCount() - 1)
		else
			if gg.getResultsCount() == 5 then
				gg.getResults(gg.getResultsCount() - 2)
			else
				if gg.getResultsCount() == 6 then
					gg.getResults(gg.getResultsCount() - 3)
				else
					if gg.getResultsCount() == 7 then
						gg.getResults(gg.getResultsCount() - 3)
					else
						if gg.getResultsCount() > 8 then
							gg.getResults(gg.getResultsCount() - 4)
						else
							gg.getResults(gg.getResultsCount() - 0)
						end
					end
				end
			end
		end
		gg.editAll("1", gg.TYPE_DWORD)
	end
	gg.clearResults()
end

function unlockall()
	choosemode()

	number = gg.prompt({ "Select the number of weapons to be disarmed. [1; 1151]" }, nil, { "number" })
	count = tonumber(number[1])

	i = 0
	while i < count do
		i = i + 1
		id = ids[i]
		if not (id == "0") then
			unlock(id, 0)
		end
	end

	gg.alert(
		"Unlock has successfully finished!\nYou should disconnect from WiFi once, and then reconnect for it to work better!"
	)
	os.exit()
end
