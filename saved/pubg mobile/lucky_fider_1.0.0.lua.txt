function cxb6()

		"20倍加速开启",
		"20倍加速关闭",
		"人物加速",
	}, nil, "")
	if SN == 1 then
		bba()
	end

	end

end

function bna()

	gg.setRanges(gg.REGION_ANONYMOUS)
	gg.searchNumber(
		"1.4012985e-45;5.6051939e-45;1;3.5873241e-43;1200;0.5;120::",
		gg.TYPE_FLOAT,
		false,
		gg.SIGN_EQUAL,
		0,
		-1
	)

		gg.toast("人物加速开启失败")

		gg.searchNumber("1F", gg.TYPE_FLOAT, false, gg.SIGN_EQUAL, 0, -1)
